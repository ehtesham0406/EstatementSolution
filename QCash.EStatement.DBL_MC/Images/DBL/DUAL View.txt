CREATE VIEW [dbo].[statement_DUAL]
AS
SELECT     dbo.STATEMENT_INFO.BANK_CODE, dbo.STATEMENT_INFO.TITLE + ' ' + dbo.STATEMENT_INFO.CLIENTNAME AS CLIENT, dbo.STATEMENT_INFO.JOBTITLE, dbo.STATEMENT_INFO.ADDRESS, 
                      dbo.STATEMENT_INFO.CITY, dbo.STATEMENT_INFO.COUNTRY, dbo.STATEMENT_INFO.ZIP, dbo.STATEMENT_INFO.PAN, dbo.STATEMENT_INFO.ACURN, 
                      CASE WHEN dbo.STATEMENT_INFO.ACURN = 'BDT' THEN dbo.STATEMENT_INFO.CRD_LIMIT ELSE round(dbo.STATEMENT_INFO.CRD_LIMIT, 0, 1) END AS CRD_LIMIT, 
                      CASE WHEN dbo.STATEMENT_INFO.ACURN = 'BDT' THEN dbo.STATEMENT_INFO.AVAIL_CRD_LIMIT ELSE round(dbo.STATEMENT_INFO.AVAIL_CRD_LIMIT, 0, 1) END AS AVAIL_CRD_LIMIT, 
                      CASE WHEN dbo.STATEMENT_INFO.ACURN = 'BDT' THEN dbo.STATEMENT_INFO.AVAIL_CASH_LIMIT ELSE round(dbo.STATEMENT_INFO.AVAIL_CASH_LIMIT, 0, 1) END AS AVAIL_CASH_LIMIT, 
                      CEILING(dbo.STATEMENT_INFO.MIN_AMOUNT_DUE) AS MIN_AMOUNT_DUE, dbo.STATEMENT_INFO.SBALANCE, dbo.STATEMENT_INFO.SUM_PURCHASE, 
                      dbo.STATEMENT_INFO.SUM_WITHDRAWAL, dbo.STATEMENT_INFO.SUM_INTEREST, dbo.STATEMENT_INFO.SUM_OTHER, dbo.STATEMENT_INFO.SUM_CREDIT, 
                      dbo.STATEMENT_INFO.OVLFEE_AMOUNT, dbo.STATEMENT_INFO.OVDFEE_AMOUNT, dbo.STATEMENT_INFO.CASH_LIMIT, dbo.STATEMENT_INFO.SUM_REVERSE, 
                      CASE WHEN (dbo.STATEMENT_INFO.EBALANCE < 0) THEN (- 1 * (CEILING(- 1 * (dbo.STATEMENT_INFO.EBALANCE)))) ELSE CEILING(dbo.STATEMENT_INFO.EBALANCE) END AS EBALANCE, 
                      dbo.STATEMENT_INFO.NEXT_STATEMENT_DATE, dbo.STATEMENT_INFO.PAYMENT_DATE, dbo.STATEMENT_INFO.STATEMENT_DATE, dbo.STATEMENT_DETAILS.TRNDATE, 
                      dbo.STATEMENT_DETAILS.POSTDATE, UPPER(dbo.STATEMENT_DETAILS.TRNDESC) AS TRNDESC, dbo.STATEMENT_DETAILS.AMOUNT, dbo.STATEMENT_DETAILS.APPROVAL, 
                      dbo.STATEMENT_INFO.EMAIL, dbo.STATEMENT_INFO.STM_MSG, dbo.STATEMENT_INFO.STARTDATE, dbo.STATEMENT_INFO.ENDDATE, dbo.STATEMENT_DETAILS.AMOUNTSIGN, 
                      dbo.STATEMENT_INFO.CONTRACTNO, dbo.STATEMENT_DETAILS.OC, dbo.STATEMENT_DETAILS.ORGAMOUNT, dbo.STATEMENT_INFO.IDCLIENT, 
                      dbo.BonusContrAcc.SUM_CREDIT AS SUM_CREDIT_FORBONUS, dbo.BonusContrAcc.SUM_DEBIT AS SUM_DEBIT_FORBONUS, dbo.BonusContrAcc.EBALANCE AS EBALANCE_FORBONUS, 
                      dbo.BonusContrAcc.SBALANCE AS SBALANCE_FORBONUS, dbo.STATEMENT_DETAILS.AutoID, dbo.STATEMENT_DETAILS.FR, dbo.ACCUM_INTERESRT_VW.ACCUM_INT_RRELEASE, 
                      dbo.ACCUM_INTERESRT_VW.ACCUM_INT_EBALANCE, dbo.ACCUM_INTERESRT_VW.ACCUM_INT_SBALANCE, dbo.ACCUM_INTERESRT_VW.ACCUM_INT_AMOUNT, 
                      dbo.ACCUM_INTERESRT_VW.ACCOUNT_NO, CASE WHEN (dbo.STATEMENT_DETAILS.P = '' AND dbo.STATEMENT_DETAILS.DOCNO != '') 
                      THEN dbo.STATEMENT_INFO.PAN ELSE dbo.STATEMENT_DETAILS.P END AS P, dbo.STATEMENT_DETAILS.DE, dbo.STATEMENT_DETAILS.DOCNO, dbo.STATEMENT_DETAILS.NO, 
                      dbo.STATEMENT_INFO.INSTALL_UNPAID_AMOUNT, dbo.STATEMENT_INFO.INSTALL_MONTH_PAYM, dbo.STATEMENT_INFO.INDICATOR, dbo.STATEMENT_INFO.TITLE, 
                      dbo.STATEMENT_INFO.PPROMOTIONALTEXT
FROM         dbo.STATEMENT_INFO LEFT OUTER JOIN
                      dbo.ACCUM_INTERESRT_VW ON dbo.STATEMENT_INFO.STATEMENTNO = dbo.ACCUM_INTERESRT_VW.StatementNo AND 
                      dbo.STATEMENT_INFO.ACURN = dbo.ACCUM_INTERESRT_VW.ACURN LEFT OUTER JOIN
                      dbo.BonusContrAcc ON dbo.STATEMENT_INFO.STATEMENTNO = dbo.BonusContrAcc.STATEMENTNO LEFT OUTER JOIN
                      dbo.STATEMENT INNER JOIN
                      dbo.STATEMENT_DETAILS ON dbo.STATEMENT.STATEMENTNO = dbo.STATEMENT_DETAILS.STATEMENTNO AND dbo.STATEMENT.CONTRACTNO = dbo.STATEMENT_DETAILS.CONTRACTNO ON 
                      dbo.STATEMENT_INFO.STATEMENTNO = dbo.STATEMENT_DETAILS.STATEMENTNO AND dbo.STATEMENT_INFO.CONTRACTNO = dbo.STATEMENT_DETAILS.CONTRACTNO AND 
                      dbo.STATEMENT_INFO.ACCOUNTNO = dbo.STATEMENT_DETAILS.ACCOUNTNO
WHERE     (dbo.STATEMENT_DETAILS.TRNDESC NOT IN ('Charge interest for 0 operations group', 'Interest On Interest', 'Interest On Service Fee', 'Interest On EMI', 'Interest On ATM', 
                      'Interest On POS', 'Interest On Previous Balance', 'Charge interest for 5', 'Charge interest for 8', 'Charge interest for 7', 'Charge interest for Installment', 'Charge interest for Installment for acceleration repayment', 
                      'Charge interest for Installment for acceleration repayment')) OR
                      (dbo.STATEMENT_DETAILS.TRNDESC IS NULL)

GO
